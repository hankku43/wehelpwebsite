module polynumber(num) {
for (i = [0 : num-1])
translate([ 0, uy*i, 0 ]) children(0);
}
module coner (x,y,H,m) {
rotate([90,0,90])polynumber(H){ unit(m);}
translate([x,0,0]) mirror([1,0,0]) rotate([90,0,90])polynumber(H){ unit(m);}
translate([0,y,0]) mirror([0,1,0]) rotate([90,0,90])polynumber(H){ unit(m);}
translate([x,y,0]) rotate([90,0,270])polynumber(H){ unit(m);}
}

module frame(xn,yn,xm,ym){
translate([T+T,T,ux]) mirror([1,0,0]) rotate([270,90,0]) polynumber(xn)

difference(){

點擊此處以 展開/收合 案例二程式碼與說明

    以下是我的程式碼，其運作原理如下。 首先以readlines()將檔案讀入字串數列。 以grep()抓取基因序列檔案中為標題行的序數， 利用gsup()編輯兩檔案中的基因名稱使其格式一致， 再利用for迴圈搭配match()逐行比對兩檔案有無對不上的基因功能。 以paste()將"funcion no found"字串加入基因名稱後方， 並將該基因名稱額外寫入新文字文件儲存， 若有比對到功能，則將功能描述文字加進序列檔案 ，並以同樣之原理將查無序列的基因功能和名稱， 額外寫入新文字文件儲存。

# 讀取fasta檔案
setwd("C:\\Users\\USER\\Desktop\\Dadabase\\populus")
list.files()
fasta <- readLines("Ptrichocarpa_533_v4.1.protein.fa")
protein_function <- readLines("Ptrichocarpa_533_v4.1.defline.txt")
modi_pro_func<- gsub("\t"," ",protein_function)
pro_fun_onlyname <- gsub("\\s.*","",protein_function)
pro_name_fasta<- fasta[grep(">",fasta)]
namelocal<-grep(">",fasta)
notfoundfunnum<-c()

u<-1

for (i in c(1:length(pro_name_fasta))) {

finding_1<-gsub("\\s.*","",fasta[namelocal[i]])
finding_2<-gsub("^>","",finding_1)
finding_3<-gsub("\\.p$", "", finding_2)
index <- match(finding_3,pro_fun_onlyname)
if(is.na(index)){
fasta[namelocal[i]] <- paste(gsub("\\s.*","",fasta[namelocal[i]]),"founction not found",sep=" ")
notfoundfunnum[u]<-namelocal[i]
u <- u+1
} else{
onlyfunindex <- sub("^\\S+\\s", "",modi_pro_func[index])
fasta[namelocal[i]] <- paste(finding_1,onlyfunindex,sep=" ")
}
}


head(fasta)
writeLines(fasta,"new_fasta.fa")
write.table(fasta[notfoundfunnum], file = "Not_Found_Function.txt", row.names = FALSE, col.names = FALSE, quote = FALSE)
purenamefasta<-gsub("^>","",gsub("\\s.*","",fasta[grep(">",fasta)]))
purenamefasta<-gsub("\\.p$", "", purenamefasta)
s<-1
Not_Found_protein <- c()
for(t in c(1:length(pro_fun_onlyname))){
if(is.na(match(pro_fun_onlyname[t],purenamefasta))){
Not_Found_protein[s]<-modi_pro_func[t]
s<-s+1
}
}
head(Not_Found_protein)
write.table(Not_Found_protein, file = "Not_Found_Protein.txt", row.names = FALSE, col.names = FALSE, quote = FALSE)
